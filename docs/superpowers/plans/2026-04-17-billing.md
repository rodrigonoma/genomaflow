# Billing: Mensalidade + Créditos + Token Tracking — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implementar billing híbrido completo — ledger de créditos, webhooks Stripe/Mercado Pago, débito por agente no worker, token tracking em `clinical_results`, dashboard de billing Angular com relatório de consumo.

**Architecture:** `credit_ledger` append-only é a fonte de verdade. Webhooks de gateway creditam via transação atômica com idempotência em `payment_events`. Worker debita 1 crédito por agente e verifica alerta de 20%. Migration 016 adiciona `input_tokens`/`output_tokens` em `clinical_results`. View `tenant_token_usage` agrega. Rota `GET /billing/usage` serve o relatório.

**Tech Stack:** PostgreSQL 15, Fastify, BullMQ worker, Angular 17+, Stripe SDK, Mercado Pago SDK, Redis pub/sub.

```sql
-- Especialidades por tenant
CREATE TABLE IF NOT EXISTS tenant_specialties (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  agent_type TEXT NOT NULL
    CHECK (agent_type IN ('metabolic','cardiovascular','hematology',
                          'small_animals','equine','bovine',
                          'therapeutic','nutrition')),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE (tenant_id, agent_type)
);

-- Assinaturas
CREATE TABLE IF NOT EXISTS subscriptions (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  gateway TEXT NOT NULL CHECK (gateway IN ('stripe','mercadopago')),
  gateway_subscription_id TEXT NOT NULL,
  plan TEXT NOT NULL,
  status TEXT NOT NULL CHECK (status IN ('active','past_due','cancelled')),
  onboarding_bonus_pct INTEGER NOT NULL DEFAULT 30,
  recurring_credits INTEGER NOT NULL DEFAULT 0,
  current_period_end TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Ledger de créditos (append-only)
CREATE TABLE IF NOT EXISTS credit_ledger (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  amount INTEGER NOT NULL,
  kind TEXT NOT NULL
    CHECK (kind IN (
      'subscription_bonus',
      'topup',
      'topup_recurring',
      'agent_usage',
      'adjustment'
    )),
  description TEXT,
  exam_id UUID REFERENCES exams(id),
  payment_event_id UUID,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Idempotência de webhooks
CREATE TABLE IF NOT EXISTS payment_events (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  gateway TEXT NOT NULL CHECK (gateway IN ('stripe','mercadopago')),
  gateway_event_id TEXT NOT NULL,
  kind TEXT NOT NULL,
  tenant_id UUID NOT NULL REFERENCES tenants(id),
  amount_brl NUMERIC(10,2) NOT NULL,
  credits_granted INTEGER NOT NULL,
  processed_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE (gateway, gateway_event_id)
);

-- View de saldo atual
CREATE OR REPLACE VIEW tenant_credit_balance AS
  SELECT tenant_id, COALESCE(SUM(amount), 0) AS balance
  FROM credit_ledger
  GROUP BY tenant_id;
```

```javascript
                COUNT(*) AS agents_executed,
                COALESCE(SUM(amount * -1), 0) AS credits_consumed

        `SELECT COALESCE(SUM(input_tokens), 0) AS input_tokens,

      // Convertido: input R$0.026/1k, output R$0.13/1k (usando câmbio ~5x)
      const estimatedCostBrl = ((inputTokens * 0.026) + (outputTokens * 0.13)) / 1000;

      return {
        period_days: days,
        exams_processed: Number(examsRes.rows[0].exams_processed),
        agents_executed: Number(examsRes.rows[0].agents_executed),
        credits_consumed: Number(examsRes.rows[0].credits_consumed),
        input_tokens: inputTokens,
        output_tokens: outputTokens,
        estimated_api_cost_brl: Math.round(estimatedCostBrl * 100) / 100
      };
    } finally {
      client.release();
    }
  });
};
```

```javascript
'use strict';

const Stripe = require('stripe');

module.exports = async function webhookRoutes(fastify) {

  // Função compartilhada: processar evento de billing
  async function processPaymentEvent(client, {
    gateway, gateway_event_id, kind, tenant_id, amount_brl, credits_granted, metadata
  }) {
    // Idempotência: verificar se já foi processado
    const existing = await client.query(
      'SELECT id FROM payment_events WHERE gateway = $1 AND gateway_event_id = $2',
      [gateway, gateway_event_id]
    );
    if (existing.rows.length > 0) return { already_processed: true };

    await client.query('BEGIN');
    try {
      // Inserir payment_event
      const evRes = await client.query(
        `INSERT INTO payment_events
           (gateway, gateway_event_id, kind, tenant_id, amount_brl, credits_granted)
         VALUES ($1, $2, $3, $4, $5, $6) RETURNING id`,
        [gateway, gateway_event_id, kind, tenant_id, amount_brl, credits_granted]
      );
      const payment_event_id = evRes.rows[0].id;

      // Creditar no ledger
      await client.query(
        `INSERT INTO credit_ledger
           (tenant_id, amount, kind, description, payment_event_id)
         VALUES ($1, $2, $3, $4, $5)`,
        [tenant_id, credits_granted, kind === 'subscription_created' ? 'subscription_bonus' : kind === 'subscription_renewed' ? 'topup_recurring' : 'topup',
         `Gateway: ${gateway} — ${kind}`, payment_event_id]
      );

      // Marcar tenant como ativo (em subscription_created)
      if (kind === 'subscription_created') {
        await client.query('UPDATE tenants SET active = true WHERE id = $1', [tenant_id]);
        // Inserir especialidades do metadata
        if (metadata?.specialties && Array.isArray(metadata.specialties)) {
          for (const agent_type of metadata.specialties) {
            await client.query(
              'INSERT INTO tenant_specialties (tenant_id, agent_type) VALUES ($1, $2) ON CONFLICT DO NOTHING',
              [tenant_id, agent_type]
            );
          }
        }
      }

      await client.query('COMMIT');

      // Publicar notificação via Redis
      if (fastify.redis) {
        await fastify.redis.publish(`billing:topup:${tenant_id}`, JSON.stringify({ balance_delta: credits_granted }));
      }

      return { ok: true, credits_granted };
    } catch (err) {
      await client.query('ROLLBACK');
      throw err;
    }
  }

  // POST /webhooks/stripe
  fastify.post('/webhooks/stripe', {
    config: { rawBody: true } // necessário para verificar assinatura
  }, async (request, reply) => {
    const sig = request.headers['stripe-signature'];
    const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET;

    let event;
    try {
      const stripe = new Stripe(process.env.STRIPE_SECRET_KEY || 'sk_test_placeholder');
      event = stripe.webhooks.constructEvent(request.rawBody || request.body, sig, webhookSecret || 'whsec_placeholder');
    } catch (err) {
      fastify.log.warn('Stripe webhook signature invalid:', err.message);
      // Em dev, sem secret configurado, processar diretamente
      if (process.env.NODE_ENV === 'production') {
        return reply.status(400).send({ error: 'Invalid signature' });
      }
      event = request.body;
    }

    const client = await fastify.pg.connect();
    try {
      const eventType = event.type || event.kind;

      if (eventType === 'customer.subscription.created') {
        const obj = event.data?.object || event;
        const tenant_id = obj.metadata?.tenant_id;
        const specialties = obj.metadata?.specialties ? JSON.parse(obj.metadata.specialties) : [];
        const amount_brl = 199.00; // plano starter

        // Bônus: 30% do valor em créditos (R$199 × 30% / R$0.49 ≈ 122 créditos)
        const credits_granted = Math.floor((amount_brl * 30 / 100) / 0.49);

        await processPaymentEvent(client, {
          gateway: 'stripe',
          gateway_event_id: event.id || obj.id,
          kind: 'subscription_created',
          tenant_id,
          amount_brl,
          credits_granted,
          metadata: { specialties }
        });

      } else if (eventType === 'invoice.payment_succeeded') {
        const obj = event.data?.object || event;
        const tenant_id = obj.metadata?.tenant_id || obj.subscription_details?.metadata?.tenant_id;
        const amount_brl = (obj.amount_paid || 19900) / 100;

        // Renovação: créditos recorrentes configurados
        const subRes = await client.query(
          'SELECT recurring_credits FROM subscriptions WHERE tenant_id = $1 AND status = $2',
          [tenant_id, 'active']
        );
        const credits_granted = subRes.rows[0]?.recurring_credits || 0;
        if (credits_granted > 0) {
          await processPaymentEvent(client, {
            gateway: 'stripe',
            gateway_event_id: event.id,
            kind: 'subscription_renewed',
            tenant_id,
            amount_brl,
            credits_granted,
            metadata: {}
          });
        }
      }

      return reply.status(200).send({ received: true });
    } finally {
      client.release();
    }
  });

  // POST /webhooks/mercadopago
  fastify.post('/webhooks/mercadopago', async (request, reply) => {
    // Mercado Pago envia notificações com tipo e id do recurso
    const { type, data } = request.body || {};

    fastify.log.info('MercadoPago webhook:', { type, data });

    if (type === 'payment') {
      // TODO: buscar detalhes do pagamento via MP SDK
      // Por ora, logar e retornar 200 para MP parar de reenviar
      fastify.log.info('MercadoPago payment notification received', data);
    }

    return reply.status(200).send({ received: true });
  });
};
```

```javascript
async function getBalance(tenantId, client) {
  const res = await client.query(
    'SELECT COALESCE(balance, 0) AS balance FROM tenant_credit_balance WHERE tenant_id = $1',
    [tenantId]
  );

async function checkLowCreditAlert(tenantId, client, redis) {
  const balance = await getBalance(tenantId, client);

  const grantedRes = await client.query(
    `SELECT COALESCE(SUM(amount), 0) AS granted
     FROM credit_ledger
     WHERE tenant_id = $1 AND amount > 0 AND created_at >= NOW() - INTERVAL '30 days'`,
    [tenantId]
  );
  const granted = Number(grantedRes.rows[0].granted);

  if (balance <= 0) {
    await redis.publish(`billing:exhausted:${tenantId}`, JSON.stringify({ balance }));
  } else if (granted > 0 && balance / granted <= 0.20) {
    await redis.publish(`billing:alert:${tenantId}`, JSON.stringify({ balance, granted }));
  }
}
```

```typescript
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { environment } from '../../../../environments/environment';

export interface BalanceResponse {
  balance: number;
}

export interface LedgerItem {
  id: string;
  amount: number;
  kind: string;
  description: string;
  exam_id: string | null;
  created_at: string;
}

export interface LedgerHistoryResponse {
  items: LedgerItem[];
  total: number;
  page: number;
  limit: number;
}

export interface UsageReport {
  period_days: number;
  exams_processed: number;
  agents_executed: number;
  credits_consumed: number;
  input_tokens: number;
  output_tokens: number;
  estimated_api_cost_brl: number;
}

@Injectable({ providedIn: 'root' })
export class BillingService {
  private api = environment.apiUrl;

  constructor(private http: HttpClient) {}

  getBalance(): Observable<BalanceResponse> {
    return this.http.get<BalanceResponse>(`${this.api}/billing/balance`);
  }

  getHistory(page = 1, limit = 20): Observable<LedgerHistoryResponse> {
    return this.http.get<LedgerHistoryResponse>(`${this.api}/billing/history?page=${page}&limit=${limit}`);
  }

  getUsage(days: number): Observable<UsageReport> {
    return this.http.get<UsageReport>(`${this.api}/billing/usage?days=${days}`);
  }

  topup(gateway: string, credits: number): Observable<{ checkout_url: string }> {
    return this.http.post<{ checkout_url: string }>(`${this.api}/billing/topup`, { gateway, credits });
  }
}
```

```typescript
export class BillingComponent implements OnInit {
  balance = signal<number>(0);
  balancePct = signal<number>(0);
  history = signal<LedgerItem[]>([]);
  usage = signal<UsageReport | null>(null);
  usageDays = 30;
  currentPage = signal<number>(1);
  totalPages = signal<number>(1);
  openTopup = false;
  selectedCredits: number | null = null;
  topupGateway: 'stripe' | 'mercadopago' | '' = '';
  topupLoading = signal<boolean>(false);
  topupError = signal<string>('');

  readonly creditPackages = [
    { credits: 100, price: 49.90, perCredit: '0,49' },
    { credits: 250, price: 109.90, perCredit: '0,44' },
    { credits: 500, price: 199.90, perCredit: '0,40' }
  ];

  constructor(private billingService: BillingService) {}

  ngOnInit(): void {
    this.loadBalance();
    this.loadHistory();
    this.loadUsage();
  }

  loadBalance(): void {
    this.billingService.getBalance().subscribe(({ balance }) => {
      this.balance.set(balance);
      // Estimar porcentagem baseado em máximo de 500 créditos (referência)
      this.balancePct.set(Math.min(100, Math.round((balance / 500) * 100)));
    });
  }

  loadHistory(): void {
    this.billingService.getHistory(this.currentPage(), 20).subscribe(({ items, total, limit }) => {
      this.history.set(items);
      this.totalPages.set(Math.ceil(total / limit));
    });
  }

  loadUsage(): void {
    this.billingService.getUsage(this.usageDays).subscribe(data => this.usage.set(data));
  }

  prevPage(): void {
    if (this.currentPage() > 1) {
      this.currentPage.update(p => p - 1);
      this.loadHistory();
    }
  }

  nextPage(): void {
    if (this.currentPage() < this.totalPages()) {
      this.currentPage.update(p => p + 1);
      this.loadHistory();
    }
  }

  kindLabel(kind: string): string {
    const labels: Record<string, string> = {
      subscription_bonus: 'Bônus',
      topup: 'Recarga',
      topup_recurring: 'Recarga Recorrente',
      agent_usage: 'Consumo',
      adjustment: 'Ajuste'
    };
    return labels[kind] ?? kind;
  }

  confirmTopup(): void {
    this.topupError.set('');
    if (!this.selectedCredits) return this.topupError.set('Selecione um pacote.');
    if (!this.topupGateway) return this.topupError.set('Selecione uma forma de pagamento.');
    this.topupLoading.set(true);
    this.billingService.topup(this.topupGateway, this.selectedCredits).subscribe({
      next: ({ checkout_url }) => {
        window.location.href = checkout_url;
      },
      error: (err) => {
        this.topupLoading.set(false);
        this.topupError.set(err.error?.error ?? 'Erro ao iniciar pagamento.');
      }
    });
  }
}
```
